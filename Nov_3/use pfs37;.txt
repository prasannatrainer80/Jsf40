use pfs37;

For employ table

-- Increase Basic by 5000 for name is 'Suhaas'

Update Employ 
set Basic=Basic+5000 WHERE Name='Suhaas';

-- For All 'Java' Dept employees increase basic by 1000;

update Employ
set Basic = Basic + 1000 where dept='Java';

-- Update basic as 90000 for empno 5
Update Employ set basic=90000 where empno = 5;

-- update Name as 'Karima Saleem' for Name 'Kareem'

-- Update Dept as 'QA' for Name 'Karthik'
Update Employ set Dept='QA' WHERE Name='Karthik';

-- Update Desig as 'Manager' for Empno 3 

Update Employ set Desig='Manager' WHERE Empno=3;
