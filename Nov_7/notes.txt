UNION : This operator allows you to combine 2 query results, but for that select command needs to have similar and equal no.of attributes. UNION operator excludes the duplicate records. 

select empno,ename,job from emp where job='CLERK'
UNION
select empno,ename,job from emp where job='SALESMAN';

select empno,ename,job,sal from emp where sal between 500 and 3000
UNION
select empno,ename,job,sal from emp where sal between 2000 and 5000;

UNION ALL : This operator is similar to UNION, but it includes duplicate records also in that.

select empno,ename,job from emp where job='CLERK'
UNION ALL
select empno,ename,job from emp where job='SALESMAN';

select empno,ename,job,sal from emp where sal between 500 and 3000
UNION ALL
select empno,ename,job,sal from emp where sal between 2000 and 5000;

select empno,ename,job,sal from emp where job in ('CLERK','MANAGER')
